# FEC + ARQ 實作計畫

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 在現有純 FEC 架構上疊加 NACK-based ARQ，使 30–50% 封包遺失率下 TCP 吞吐量不再歸零。

**Architecture:**
RX 端偵測到 block 解碼失敗（50 ms 逾時）後，透過既有 UDP socket 對 TX 發送 TYPE_NACK。
TX 端收到 NACK 後，從 TX Block Cache 取出原始封包，以**新的隨機係數**重新 encode，發送 k+1 個修復 shard。
TX Block Cache 為環狀 buffer（64 槽），按 3×RTT（fallback 300 ms）驅逐。

**Tech Stack:** C99, POSIX sockets, 現有 GF(2⁸)/codec/transport 模組，無新依賴。

---

## 現況問題

E2 量測（E2 results）顯示：
- 0% loss → 101 Mbps；10% → 10 Mbps；20% → 2 Mbps；30–50% → ~0 Mbps
- 純 FEC 在 k=2、n=3 時只能容忍 ≤1 shard 遺失；>33% loss 時 TCP 停頓

ARQ 目標：在 40–50% loss 下仍能恢復 TCP 連線（吞吐量 >0）。

```

在 `transport_send_probe_echo()` 之後新增宣告：

 * Send a NACK for block_id on path[path_idx].
 * RX calls this when a block expires undecoded.
 */

                        int path_idx,
                        uint32_t block_id);
```

**Step 2: 在 transport.c 實作 transport_send_nack()**

仿照 `transport_send_probe()` 的模式，header 填 TYPE_NACK，block_id 填目標 block，payload_len=0：

```c
int transport_send_nack(struct transport_ctx *ctx, int path_idx,
                        uint32_t block_id)
{
    uint8_t buf[WIRE_HDR_SIZE];
    struct wire_header *hdr = (struct wire_header *)buf;

    memset(hdr, 0, WIRE_HDR_SIZE);
    hdr->magic    = htons(WIRE_MAGIC);
    hdr->version  = WIRE_VERSION;
    hdr->type     = TYPE_NACK;
    hdr->block_id = htonl(block_id);
    return send_buf(ctx, path_idx, buf, WIRE_HDR_SIZE);
}
```

**Step 3: 在 transport_recv() 加入 TYPE_NACK 處理**

在 `transport_recv()` 的最後（`return -1;` 之前），加入：

```c
if (hdr->type == TYPE_NACK) {
    /* block_id already byte-swapped above; no payload to parse */
    return TYPE_NACK;
}
```

（注意：`hdr->block_id` 在 `transport_recv()` 頂端的 `ntohl()` 已處理。caller 從 `hdr->block_id` 讀 missing block id。）

### Part B — TX Block Cache

**Step 4: main.c 加入 TX Block Cache 結構（在 struct tx_block 下方）**

```c
/* -------------------------------------------------------------------------
 * TX Block Cache: stores recent blocks for NACK-triggered retransmission.
 * Circular buffer; slots evicted after 3 * RTT or ARQ_CACHE_TTL_US.
 * ---------------------------------------------------------------------- */
#define ARQ_CACHE_SIZE_DEFAULT  64
#define ARQ_CACHE_TTL_US        300000ULL   /* 300 ms fallback if RTT unknown */

struct tx_cache_entry {
    uint32_t block_id;
    uint8_t  pkts[MAX_K][MAX_PAYLOAD];
    uint16_t pkt_len[MAX_K];
    int      pkt_count;
    uint64_t cached_at_us;
    bool     valid;
};

struct tx_block_cache {
    struct tx_cache_entry *slots;   /* dynamically allocated */
    int                    size;
    int                    write_idx;
};
```

**Step 5: 加入 cache helper functions（在 flush_block 之前）**

```c
static void tx_cache_init(struct tx_block_cache *c, int size)
{
    c->slots     = (struct tx_cache_entry *)calloc((size_t)size,
                                                    sizeof(*c->slots));
    c->size      = size;
    c->write_idx = 0;
}

static void tx_cache_free(struct tx_block_cache *c)
{
    free(c->slots);
    c->slots = NULL;
}

static void tx_cache_store(struct tx_block_cache *c, const struct tx_block *blk)
{
    struct tx_cache_entry *e = &c->slots[c->write_idx];
    int i;
    e->block_id      = blk->block_id;
    e->pkt_count     = blk->pkt_count;
    e->cached_at_us  = now_us();
    e->valid         = true;
    for (i = 0; i < blk->pkt_count; i++) {
        memcpy(e->pkts[i], blk->pkts[i], blk->pkt_len[i]);
        e->pkt_len[i] = blk->pkt_len[i];
    }
    c->write_idx = (c->write_idx + 1) % c->size;
}

static struct tx_cache_entry *tx_cache_find(struct tx_block_cache *c,
                                             uint32_t block_id)
{
    int i;
    for (i = 0; i < c->size; i++) {
        if (c->slots[i].valid && c->slots[i].block_id == block_id)
            return &c->slots[i];
    }
    return NULL;
}

static void tx_cache_evict(struct tx_block_cache *c, uint64_t max_age_us)
{
    uint64_t now = now_us();
    int i;
    for (i = 0; i < c->size; i++) {
        if (c->slots[i].valid &&
            (now - c->slots[i].cached_at_us) >= max_age_us)
            c->slots[i].valid = false;
    }
}
```

在 `bool expired = ...` 判斷邏輯替換為：

```c
bool expired = false;
if (!win->slots[i].decoded) {
    uint64_t age = now_us() - win->slots[i].first_recv_us;
    if (!win->slots[i].nack_sent) {
        /* First timeout: send NACK and give block another window */
        if (age >= RX_BLOCK_TIMEOUT_US) {
            if (arq_enabled && tctx) {
                /* Send NACK on path 0; RX doesn't know best path */
                transport_send_nack(tctx, 0, win->slots[i].block_id);
                LOG_DBG("NACK sent for block %u", win->slots[i].block_id);
            }
            win->slots[i].nack_sent    = true;
            win->slots[i].nack_sent_us = now_us();
            /* Reset first_recv_us to give block another RX_BLOCK_TIMEOUT_US */
            win->slots[i].first_recv_us = now_us();
        }
    } else {
        /* Already NACKed: evict after second timeout */
        uint64_t nack_age = now_us() - win->slots[i].nack_sent_us;
        if (nack_age >= RX_BLOCK_TIMEOUT_US)
            expired = true;
    }
}
```

（`decoded || expired` 的 eviction 邏輯維持不變。）

### Part B — TX: NACK Handler

**Step 4: 新增 handle_nack() 函式（在 flush_block 之後）**

```c
/* -------------------------------------------------------------------------
 * Handle incoming NACK: re-encode block from cache and send repair shards.
 * ---------------------------------------------------------------------- */
static void handle_nack(struct tx_block_cache *cache,
                        struct transport_ctx  *tctx,
                        struct strategy_ctx   *sctx,
                        uint32_t block_id, int k)
{
    struct tx_cache_entry *e;
    struct shard           shards[MAX_N];
    int                    n_repair, i, path;
    uint16_t               max_len = 0;
    int                    j;

    e = tx_cache_find(cache, block_id);
    if (!e) {
        LOG_DBG("NACK for block %u: not in cache", block_id);
        return;
    }

    /* Pad to uniform length (same as flush_block does) */
    for (j = 0; j < e->pkt_count; j++)
        if (e->pkt_len[j] > max_len) max_len = e->pkt_len[j];
    for (j = 0; j < e->pkt_count; j++) {
        if (e->pkt_len[j] < max_len)
            memset(e->pkts[j] + e->pkt_len[j], 0,
                   (size_t)(max_len - e->pkt_len[j]));
    }

    /* Use k+1 repair shards (same k, new random coefficients) */
    n_repair = k + 1;
    if (n_repair > MAX_N) n_repair = MAX_N;

    encode_block((const uint8_t (*)[MAX_PAYLOAD])e->pkts,
                 e->pkt_len, k, n_repair, shards);

    for (i = 0; i < n_repair; i++) {
        path = strategy_next_path(sctx);
        if (path < 0) break;
        if (transport_send_shard(tctx, path, block_id,
                                 (uint8_t)i, (uint8_t)k, (uint8_t)n_repair,
                                 &shards[i]) != 0)
            LOG_WARN("NACK retransmit failed: block=%u shard=%d", block_id, i);
    }
    LOG_DBG("NACK handled: block=%u sent %d repair shards", block_id, n_repair);
}
```

## 常見陷阱

- **`struct wire_header` packed**：`block_id` 欄位 network byte order，`transport_recv()` 已做 `ntohl()`，caller 直接讀 `hdr.block_id` 即可。
- **NACK 送到哪個 path**：RX 沒有 path quality 資訊，固定送 path 0（在單路徑測試環境下無影響）。多路徑未來優化可廣播所有 path。
- **cache 大小 vs. block rate**：iperf3 每 ~11 ms 產生一個 block（10 Mbps），64 slots × 300 ms TTL = ~27 blocks，足夠。調低 `arq_cache_size` 可減 memory。
- **NACK storm**：若 TX cache miss（block 太舊），`handle_nack()` 靜默忽略，不重送，避免放大 NACK。
- **race on rx_block eviction**：`nack_sent` 後 `first_recv_us` 被重置，第二次 timeout 由 `nack_sent_us` 計算，避免雙重驅逐。
